CREATE TABLE "MEMBER" (
   "M_NO"   NUMBER      PRIMARY KEY,
   "TYPE"   NUMBER      REFERENCES "MEMBER_TYPE",
   "EMAIL"   VARCHAR2(50)      NOT NULL		UNIQUE,
   "PW"   VARCHAR2(1000)      NOT NULL,
   "NICKNAME"   VARCHAR2(30)      NOT NULL	UNIQUE,
   "UPDATE_DATE"   DATE   DEFAULT SYSDATE   NOT NULL,
   "ENROLL_DATE"   DATE   DEFAULT SYSDATE   NOT NULL,
   "STATUS"   VARCHAR2(2)   DEFAULT 'Y'   NOT NULL,
   "D_COUNT"   NUMBER   DEFAULT 0   NOT NULL
);

CREATE TABLE "CO_INFO" (
   "M_NO"   NUMBER      REFERENCES "MEMBER" PRIMARY KEY,
   "REGNUM"   VARCHAR2(100)      NOT NULL,
   "FILE"   VARCHAR2(100)      NOT NULL,
   "PATH"   VARCHAR2(100)      NOT NULL,
   "NAME"   VARCHAR2(100)      NOT NULL,
   "PHONE"   VARCHAR2(100)      NOT NULL,
   "ADDRESS"   VARCHAR2(100)      NOT NULL,
   "CEO"   VARCHAR2(100)      NOT NULL,
   "CO_PHONE"   VARCHAR2(100)      NOT NULL,
   "DESCRIPT"   VARCHAR2(4000)      NULL,
   "BIRTH_DATE"   DATE      NULL,
   "MEMSUM"   NUMBER      NULL,
   "REVENUE"   NUMBER      NULL,
   "HISTORY"   VARCHAR2(4000)      NULL,
   "WELFAIR"   VARCHAR2(4000)      NULL
);

CREATE TABLE "BOARD" (
   "B_NO"   NUMBER      PRIMARY KEY,
   "M_NO"   NUMBER      REFERENCES "MEMBER",
   "HEAD"   VARCHAR2(100)      NOT NULL,
   "TITLE"   VARCHAR2(100)      NOT NULL,
   "CONTENTS"   VARCHAR2(4000)      NOT NULL,
   "ENROLL_DATE"   DATE   DEFAULT SYSDATE   NOT NULL,
   "UPDATE_DATE"   DATE   DEFAULT SYSDATE   NOT NULL,
   "FILE_ORINAME"   VARCHAR2(100)      NULL,
   "TIME"   VARCHAR2(100)      NULL,
   "COUNT"   NUMBER   DEFAULT 0   NOT NULL,
   "STATUS"   VARCHAR2(2)   DEFAULT 'Y'   NOT NULL,
   "D_COUNT"   NUMBER   DEFAULT 0   NOT NULL,
   "EDITFILE"   VARCHAR2(100)      NULL,
   "PATH"   VARCHAR2(100)      NULL,
   "DOWN_COUNT" NUMBER DEFAULT 0 NOT NULL,
   "REPLY_COUNT" NUMBER DEFAULT 0 NOT NULL
);

CREATE TABLE "REPLY" (
   "RE_NO"   NUMBER      PRIMARY KEY,
   "M_NO"   NUMBER      REFERENCES "MEMBER",
   "B_NO"   NUMBER      REFERENCES BOARD,
   "ENROLL_DATE"   DATE   DEFAULT SYSDATE   NOT NULL,
   "UPDATE_DATE"   DATE   DEFAULT SYSDATE   NOT NULL,
   "CONTENTS"   VARCHAR2(1000)      NOT NULL,
   "STATUS"   VARCHAR2(2)   DEFAULT 'Y'   NOT NULL,
   "D_COUNT"   NUMBER   DEFAULT 0   NOT NULL
);

CREATE TABLE "NOTIFICATION" (
   "NOTI_NO"   NUMBER      PRIMARY KEY,

   "TITLE"   VARCHAR2(50)      NOT NULL,
   "END_DATE"   DATE      NOT NULL,
   "END_TIME"   DATE      NOT NULL,
   "ENROLL_DATE"   DATE   DEFAULT SYSDATE   NOT NULL,
   "P_LANGUAGE"   VARCHAR2(1000)      NOT NULL,
   "SALARY"   NUMBER      NOT NULL,
   "CONTENTS"   VARCHAR2(4000)      NOT NULL,
   "JOBS"   VARCHAR2(30)      NOT NULL,
   "HOPE"   VARCHAR2(1000)      NOT NULL,
   "STATUS"   VARCHAR2(2)   DEFAULT 'N'   NOT NULL,
   "D_COUNT"   NUMBER   DEFAULT 0   NOT NULL,
   "OPEN" VARCHAR2(2) DEFAULT 'N' NOT NULL
);
